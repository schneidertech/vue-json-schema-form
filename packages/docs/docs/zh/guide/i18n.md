*

 schema

::: tip
 `ui-schema`

 `schema` JSON Schema
:::



::: demo
```html
<template>
    <vue-form
        v-model="formData"
        :schema="schema"
        :form-footer="formFooter"
        :ui-schema="uiSchema"
    >
    </vue-form>
</template>
<script>
//
const transKeyMap = {
    userName: 'Username'
};

//
const trans = key => transKeyMap[key] || `transKeyMap.${key}`;

export default {
    name: 'Demo',
    data() {
        return {
            formData: {},
            schema: {
                type: 'object',
                required: [
                    'userName'
                ],
                properties: {
                    userName: {
                        type: 'string',
                        title: ''
                    }
                }
            },
            formFooter: {
                okBtn: 'Save', //
                cancelBtn: 'Cancel' //
            },
            uiSchema: {
                userName: {
                    'ui:options': {
                        title: trans('userName')
                    },
                }
            },
        };
    }
};
</script>
```
